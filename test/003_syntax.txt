// Testing the parser

extern void printf(u8 fmt, u8 a)

// print out number
void print(u8 number) {
  printf("%d\n", number);
}

void printXTimes(u8 count) {
  u8 i = count;
  while (i > 0) {
    print(count);
    i = i - 1;
  }
}

void main() {
  u8 one = 1;
  u8 two = 2;
  print(two);
  printXTimes(one + two);
}
